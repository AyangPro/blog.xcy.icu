    部署步骤

    # 部署步骤

    cd /opt/jinliyuzhou/server/app    # 进入程序文件目录
ll                                # 查看目录下程序文件
ps -ef | grep <程序文件名>         # 查看程序的进程号
kill -9 <进程号>                   # 杀死程序运行的进程
rm -rf <程序文件名>                # 移除程序文件
                                  # 上传新的程序文件
nohup java -jar <程序文件名> &     # 运行新的程序文件
tail -f nohup                     # 通过查看日志判断程序是否启动成功
